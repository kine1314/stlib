void MergeList(SqList La, SqList Lb, SqList *Lc)
{
	int i = 1, j = 1, k = 0;
	int La_len, Lb_len;
	ElemType ai, bj;
	
	InitList(Lc);	// ´´½¨¿Õ±íLc
	La_len = ListLength(La);
	Lb_len = ListLength(Lb);
	while(i <= La_len && j <= Lb_len)	// ±íLaºÍ±íLb¾ù·Ç¿Õ
	{
		GetElem(La, i, &ai);
		GetElem(Lb, j, &bj);
		if(ai <= bj)		// ai¸üĞ¡½«ai²åÈëLcÖĞ
		{
			ListInsert(Lc, ++k, ai);
			++i;
		}
		else				// bj¸üĞ¡½«bj²åÈëLcÖĞ
		{
			ListInsert(Lc, ++k, bj);
			++j;
		}
	}

	// ±íLa·Ç¿ÕÇÒ±íLb¿ÕÔò½«LaµÄÊ£Óà²¿·Ö½Óµ½LcÖĞ
	while(i <= La_len)
	{
		GetElem(La, i++, &ai);
		ListInsert(Lc, ++k, ai);
	}
	// ±íLb·Ç¿ÕÇÒ±íLa¿Õ Ôò½«LbµÄÊ£Óà²¿·Ö½Óµ½LcÖĞ
	while(j <= Lb_len)
	{
		GetElem(Lb, j++, &bj);
		ListInsert(Lc, ++k, bj);
	}
}

// ²éÕÒ±íÖĞÑ§ºÅÎªnumµÄ½áµã,ÈçÕÒµ½,qÖ¸Ïò´Ë½áµã,pÖ¸ÏòqµÄÇ°Çı,

Status FindFromNum(LinkList L,long num,LinkList *p,LinkList *q)
{
	*p = L;
	while(*p)
	{
		*q = (*p)->next;
		if(*q && (*q)->data.num > num) // ÒòÎªÊÇ°´Ñ§ºÅ·Ç½µĞòÅÅÁĞ
			break;
		if(*q && (*q)->data.num == num) // ÕÒµ½¸ÃÑ§ºÅ
			return TRUE;
		*p = *q;
	}
	return FALSE;
}

// ²éÕÒ±íÖĞĞÕÃûÎªnameµÄ½áµã,ÈçÕÒµ½,qÖ¸Ïò´Ë½áµã,pÖ¸ÏòqµÄÇ°Çı,

Status FindFromName(LinkList L,char name[],LinkList *p,LinkList *q)
{
	*p = L;
	while( *p )
	{
		*q = (*p)->next;
		if(*q && !strcmp((*q)->data.name, name)) // ÕÒµ½¸ÃĞÕÃû
			return TRUE;
		*p = *q;
	}
	return FALSE;
}

		p->next = q->next;

	}
	return FALSE;
}
